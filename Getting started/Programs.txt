num1 = int(input("Enter number one : "))

if op == "+":
    a = num1 + num2
    print(f"Answer for Addition of {num1} and {num2} is : ", a)
elif op == "-":
    s = num1 - num2
    print(f"Answer for Substraction of {num1} and {num2} is : ", s)
elif op == "*"  : 
    m = num1 * num2
    print(f'Answer for Multiplication of {num1} and {num2} is : ',m)

    d = num1 / num2
